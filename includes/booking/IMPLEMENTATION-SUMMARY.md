## 🔧 توابع اصلی

```php
// دریافت تنظیمات
$settings = MR_Booking::get_doctor_settings($doctor_id);

// ذخیره‌سازی
MR_Booking::save_doctor_settings($doctor_id, $settings);

// دریافت ساعات موجود
$times = MR_Booking::get_available_times($doctor_id, $date, 30);

// محاسبۂ قیمت
$final = MR_Booking::calculate_final_price($base_price, $code);

// دریافتٔ دکترهای موجود
$doctors = MR_Booking::get_available_doctors($date);
```
